CREATE TABLE orders (
    id INT PRIMARY KEY NOT NULL AUTO_INCREMENT,
    ordered_at DATETIME,
    delivered_at DATETIME,
    item_count INT NOT NULL,
    customer_id INT, -- This is the foreign key
    FOREIGN KEY (customer_id) REFERENCES customer(id)
);
